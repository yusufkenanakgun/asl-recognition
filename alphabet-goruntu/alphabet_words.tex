% Letter cell: square image + tiny label
\newcommand{\letterbox}[2]{%
  \begin{minipage}[t]{0.22\textwidth}
    \centering
    \fbox{\includegraphics[width=4.0cm,height=4.0cm,keepaspectratio]{#1}}\\[3pt]
    {\large\bfseries #2}
  \end{minipage}%
}

% Word row: three frames side-by-side under a single word label
\newcommand{\wordrow}[2]{%
  % args: word-stem (e.g. "drink"), display label
  \begin{minipage}[t]{\textwidth}
    \centering
    {\large\bfseries #2}\\[4pt]
    \includegraphics[width=0.30\textwidth,height=4.8cm,keepaspectratio]{#1_1}\hfill
    \includegraphics[width=0.30\textwidth,height=4.8cm,keepaspectratio]{#1_2}\hfill
    \includegraphics[width=0.30\textwidth,height=4.8cm,keepaspectratio]{#1_3}
  \end{minipage}%
}
